Program -> Declaration-list Main-function
Declaration-list -> Declaration Declaration-list | epsilon
Declaration -> Type-specifier ID Var-declaration-prime
Var-declaration-prime -> ; #declare-var | [ NUM ] ; #declare-array
Type-specifier -> int
Main-function -> void main ( ) Compound-stmt
Compound-stmt -> { Declaration-list Statement-list }
Statement-list -> Statement Statement-list | epsilon
Statement -> Expression-stmt | Compound-stmt | Selection-stmt | Iteration-stmt | Output-stmt
Expression-stmt -> Expression ; | ;
Output-stmt -> output ( Expression ) ; #print
Expression -> ID B | NUM | ( Expression ) | Expression + Expression #add | Expression - Expression #sub
B -> = Expression #assign | [ Expression ] #calc-arr-addr H | epsilon
H -> = Expression #assign | epsilon
Selection-stmt -> if ( Expression ) Statement else Statement
Iteration-stmt -> repeat Statement until ( Expression )
# No function call/return rules
